        async function checkFollowStatus(starId, buttonElement) {  
            if (\!isAuthReady || \!currentUserId || \!db) return;

        async function toggleFollow(starId, buttonElement) {  
            if (\!isAuthReady || \!currentUserId || \!db) { return; }  
            if (buttonElement.closest('.ad-placeholder-native')) return;
